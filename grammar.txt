program        -> declaration* EOF ;

// Top-level declarations
declaration    -> funDecl | structDecl ;
funDecl        -> "fn" type IDENTIFIER "(" parameters? ")" block ;
structDecl     -> "struct" IDENTIFIER "{" structMember* "}" ;
structMember   -> fieldDecl | methodDecl ;
fieldDecl      -> type IDENTIFIER ";" ;
methodDecl     -> "method" type IDENTIFIER "(" parameters? ")" block ;

// Types
// Type parsing consumes one type token. Built-in refs such as "int&" are
// single lexer tokens. Declared struct refs/pointers such as "Vec3&" and
// "Vec3*" are accepted as identifier-like type tokens after predeclaration.
type           -> "int"
               | "float"
               | "double"
               | "bool"
               | "null"
               | "int&" | "float&" | "double&" | "bool&"
               | DECLARED_STRUCT_TYPE
               | DECLARED_STRUCT_REF_OR_PTR ;

parameters     -> parameter ( "," parameter )* ;
parameter      -> type IDENTIFIER ;

// Statements
statement      -> echoStmt
               | block
               | ifStmt
               | whileStmt
               | returnStmt
               | exprStmt ;

block          -> "{" blockItem* "}" ;
blockItem      -> varDecl | statement ;
varDecl        -> type IDENTIFIER ( "=" expression )? ";" ;
echoStmt       -> "echo" expression ";" ;
exprStmt       -> expression ";" ;
ifStmt         -> "if" "(" expression ")" block ( "else" block )? ;
whileStmt      -> "while" "(" expression ")" block ;
returnStmt     -> "return" expression? ";" ;

// Expressions, written by precedence from lowest to highest.
expression     -> assignment ;
assignment     -> asExpr ( "=" asExpr )* ;
asExpr         -> logic_or ( "as" type )* ;
logic_or       -> logic_and ( "or" logic_and )* ;
logic_and      -> equality ( "and" equality )* ;
equality       -> comparison ( ( "!=" | "==" ) comparison )* ;
comparison     -> term ( ( ">" | ">=" | "<" | "<=" ) term )* ;
term           -> factor ( ( "-" | "+" ) factor )* ;
factor         -> unary ( ( "/" | "*" ) unary )* ;
unary          -> ( "!" | "-" ) unary | call ;
call           -> primary ( "(" arguments? ")" | "." IDENTIFIER )* ;
primary        -> "true"
               | "false"
               | "null"
               | "this"
               | INTEGER
               | FLOAT
               | DOUBLE
               | STRING_LITERAL
               | IDENTIFIER
               | "new" type "{" expression "}"
               | "(" expression ")" ;

arguments      -> expression ( "," expression )* ;
